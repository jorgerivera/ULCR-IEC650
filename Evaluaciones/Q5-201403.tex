\begin{block}{Ejercicio único}
\begin{small}
Describa, en sus propias palabras, el comportamiento que un usuario observaría si utilizara un sistema basado en Arduino, con un push button en el pin 10 y un led en el pin 11.
\end{small}
\end{block}

\vspace{-0.3in}
\begin{columns}[t]
\column{.45\textwidth}

\begin{scriptsize}
\begin{verbatim}
unsigned long final;
int estado = 1;

void setup() {
  pinMode(10, INPUT);
  pinMode(11, OUTPUT);
}
void loop() {
  maquina();
}
\end{verbatim}
\end{scriptsize}

\column{.45\textwidth}
\begin{scriptsize}
\begin{verbatim}
void maquina() {
  switch(estado) {
  case 1:
  	if (digitalRead(10)==HIGH) {
  	  estado = 2;
  	  final = millis()+5000;
  	  digitalWrite(11,HIGH);
  	}
  	break;
  case 2:
    if (millis()>=final) {
      estado = 1;
      digitalWrite(11,LOW);
    }
    break;
  }  
}
\end{verbatim}
\end{scriptsize}
