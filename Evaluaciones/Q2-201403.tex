\begin{verbatim}

void loop() {
  if (Serial.available() > 0) {
    char dato = Serial.read();
    switch (dato) {
      case 'A':
        Serial.println("Laboratorio");
        break;
      case 'B':
        Serial.println("Sistemas");
        break;
      case 'C':
        Serial.println("Digitales");
        break;
      default:
        Serial.println("Universidad Latina");
        break;
    }
  }  
}
